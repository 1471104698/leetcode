class Solution {
    public int[] getStrongest(int[] arr, int k) {
        Arrays.sort(arr);
        int len = arr.length;
        int m = arr[(len - 1) / 2];

        //è¶Šä¸­é—´çš„æ•°è¶Šé è¿‘ä¸­ä½æ•°ï¼Œå› æ­¤æˆ‘ä»¬ä»ä¸¤è¾¹æ‰¾
        int[] res = new int[k];
        int i = 0;
        int left = 0;
        int right = len - 1;
        while(k-- > 0){
            int l = Math.abs(arr[left] - m);
            int r = Math.abs(arr[right] - m);
            if(l > r){
                res[i++] = arr[left];
                left++;
                
            }else{
                res[i++] = arr[right];
                right--;
            }
        }
        return res;
    }
}
